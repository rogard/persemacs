      (&key key-list value-list result (discard nil) msg-type)
    "Recursively zip KEY-LIST and VALUE-LIST into a plist. Optionally discard extra values.

    Syntax:
    - KEY-LIST (quote (KEYWORD...))"
        (cond
         ((null key-list)
          (cond
           ((null value-list) result)
           (discard result)
           (t (error "Length mismatch: values left over: %S" value-list))))

         (t
          (let ((key (car key-list)))
            (cond
             ((not (keywordp key))
              (error "Expected keyword, got: %S" key))

              (er317/zip-to-plist
               :key-list (cdr key-list)
               :value-list (cdr value-list)
               :result (append result (list key (car value-list)))
               :discard discard
               :msg-type msg-type)))))))

    "Return a plist containing only the key-value pairs from KEYVAL-LIST
whose values are non-nil."
    (let ((result nil))
      (while keyval-list
        (let ((key (pop keyval-list))
              (value (pop keyval-list)))
          (when value
            (setq result (append result (list key value))))))
      result))

#+begin_src emacs-lisp
  (cl-defmacro er317/refactor
      (&key
       source-func
       target-key-list
       result-key-list
       target-func
       msg-type)
    "Define TARGET-FUNC as a keyword-based function that wraps around SOURCE-FUNC.

         Syntax:
         - SOURCE-FUNC              (quote SYMB) | (lambda (...) ...)
         - <RESULT|TARGET>-KEY-LIST (KEYWORD ... )
         - TARGET-FUNC              (quote SYMB)

         Compile-time:
         - RESULT-FUNC constructed using RESULT-KEY-LIST else `#'identity`

         Runtime behavior:
         TARGET-FUNC (KEYWORD VALUE ...)
         - RESULT <- (apply SOURCE-FUNC VALUE-LIST)
         - RESULT <- (RESULT-FUNC RESULT)
         - RESULT

         Example:
         (er317/refactor
         :source-func 'org-babel-get-src-block-info
         :target-func 'er317/org-babel-info-as-plist
         :target-key-list (:no-eval :datum)
         :result-key-list (:lang :body :header-args :switches :name :start :coderef))"
    (let* (
  	 (result-func
  	  (cond
  	   (result-key-list
  	    `(lambda (result-list)
  	       (er317/zip-to-plist :key-list ',result-key-list :value-list result-list)))
  	   (t #'identity)))
  	 (target-symb
            (cond
             ((symbolp target-func) target-func)
             ((and (consp target-func) (eq (car target-func) 'quote))
              (cadr target-func))
             (t (error "target-func must be a symbol or quoted symbol"))))
     	 (target-symb-list
            (cl-loop for k in target-key-list
                     collect (intern (substring (symbol-name k) 1)))))
      `(cl-defun ,target-symb (&key ,@target-symb-list)
         ,(format "Refactored wrapper around `%s`, accepting keyword args %s."
     		source-func target-key-list)
         (let* ((result-raw (apply ,source-func (list ,@target-symb-list)))
                (result (funcall ,result-func result-raw)))
     	 result))))
#+end_src

  (cl-defun er317/org-babel-comment (&key lang)
    "Return a format control string for commenting in a given mode.
  LANG is a string like \"python\", \"sh\", or \"latex\"."
    (interactive)
    (let* ((mode-fn (intern (concat lang "-mode"))))

          (funcall mode-fn)

              ;; Block-style comment format
              (format "%s %%s %s" (string-trim start) (string-trim end))))))))

  (cl-defun er317/noweb-ref-filter
      (&key
       ref-list
       (head-re "^")
       (middle-re (format "\\(%s\\)" (string-join ref-list "\\|")))
       (tail-re "$")
       (regex (concat head-re middle-re tail-re))
       (func (lambda (refactored-info) (plist-get refactored-info :body)))
       arg-list
       (babel-info-option-list '(t))
       element-parse-option-list)
    "Filters source block info by matching noweb-ref on REGEX, transforming each as FUNC (AS-PLIST-INFO ARG-LIST)."
    (let ((results '()))
      (org-element-map (apply #'org-element-parse-buffer element-parse-option-list) 'src-block
        (lambda (src)
          (let* ((begin (org-element-property :begin src))
                 (info (save-excursion
                         (goto-char begin)
                         (apply #'er317/org-babel-info-as-plist babel-info-option-list)))
                 (params (plist-get info :header-args))
                 (ref (plist-get params :noweb-ref)))
            (when (and ref (string-match-p regex ref))
              (push (apply func info arg-list) results)))))
      (nreverse results)))
