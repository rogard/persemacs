  (cl-defmacro er317/inspect-head (&key object)
    "Inspect the given Emacs Lisp OBJECT and classify its type.

  Return one of the following keywords describing the structure of OBJECT:

    Return:          OBJECT
  ----------------------------------------------------------------------------
    :lambda          lambda expression (e.g. (lambda (x) ...))
    :quoted-lambda   quoted lambda (e.g. '(lambda (x) ...))
    :quoted-symbol   quoted symbol (e.g. 'foo)
    :quoted-data     quoted data other than a symbol or lambda
    :list            a general list (function call, macro, or list literal)
    :symbol          symbol (not inside a list)
    :unknown         all other cases (numbers, strings, nil, etc.)

  The argument OBJECT is expected to be an Emacs Lisp form, which this macro analyzes
  to help determine its syntactic category."
    (if (consp object)
        (let ((head (car object))
              (body (cdr object)))
          (cond
           ((eq head 'lambda) :lambda)
           ((memq head '(quote function))
            (let ((quoted (car body)))
              (cond
               ((and (consp quoted) (eq (car quoted) 'lambda)) :quoted-lambda)
               ((symbolp quoted) :quoted-symbol)
               (t :quoted-data))))
           (t :list)))
      (cond
       ((symbolp object) :symbol)
       (t :unknown))))

      (&key

      (&key
       at-point
       datum
       name
       position
       ;;
       buffer
       no-eval
       &aux
       (__arglist `(:at-point ,at-point :buffer ,buffer :datum ,datum :name ,name :no-eval ,no-eval :position ,position))
       (__non-nil-list (cl-remove-if-not #'identity (list at-point datum position name)))
       (__has-conflict (let ((len (length __non-nil-list)))
                         (cond
  			;; setup that way bc < messes () matching
  			((eq len 0) nil)
  			((eq len 1) nil)
                          (error "Conflicting inputs: %S" __non-nil-list))))
       &allow-other-keys)
    "Resolve source block's Babel info.

  Keywords:
  - at-point|datum|name|position
  - buffer
  - no-eval

  Implemetation:
  - Forwards to `(er317/babel-info-as-plist
       :datum datum
       :no-eval no-eval)`

  `(er317/babel-info-resolve :name \"json-foo\")`"

     ((and buffer (not (eq buffer (current-buffer))))
      (with-current-buffer buffer
        (apply #'er317/babel-info-resolve __arglist)))

     (name
      (let ((new-pos (org-babel-find-named-block name))
            (new-prop (plist-put (plist-put __arglist :name nil) :position new-pos)))
        (apply #'er317/babel-info-resolve new-prop)))

     (position
      (let ((new-prop (plist-put __arglist :position nil)))
        (goto-char position)
        (apply #'er317/babel-info-resolve new-prop)))

     ((or at-point datum)
      (er317/babel-info-as-plist
       :datum datum
       :no-eval no-eval))

     (t
      (apply #'er317/babel-info-resolve

  (cl-defun er317/babel-comment (&key lang)
    "Return a format control string for commenting in a given mode.
  LANG is a string like \"python\", \"sh\", or \"latex\"."
    (interactive)
    (let* ((mode-fn (intern (concat lang "-mode"))))
      (if (not (fboundp mode-fn))
          (error "Unrecognized mode function: %s" mode-fn)
        (with-temp-buffer
          (funcall mode-fn)
          (let ((start (or comment-start ""))
                (end (or comment-end "")))
            (if (string-empty-p end)
                ;; Single-line comment format
                (format "%s %%s" (string-trim start))
              ;; Block-style comment format
              (format "%s %%s %s" (string-trim start) (string-trim end))))))))

  (cl-defun er317/fn-keyword-alist
      (&key symbol
  	  &aux
  	  (__fn (symbol-function symbol))
  	  (__docstring (and (functionp __fn) (documentation __fn t)))
  	  (__argstring (when (and __docstring
  				  (string-match "(fn\\s-+\\(.*\\))" __docstring))
  			 (match-string 1 __docstring)))
  	  (__parsed (and __argstring
  			 (read-from-string (concat "(" __argstring ")"))))
  	  (__args (car-safe __parsed))
  	  __result)
    "Extract keyword symbols and their default values from a cl-defun SYMBOL."
    (when (and __args (memq '&key __args))
      (let ((__key-args (cdr (memq '&key __args))))
        (dolist (__arg __key-args)
          (cond
           ((eq __arg '&allow-other-keys) nil)
           ((symbolp __arg)
            (push (list __arg) __result))
           ((and (listp __arg) (symbolp (car __arg)))
            (push (list (car __arg) (cadr __arg)) __result))))))
    (nreverse __result))

    "Merge SOURCE and TARGET property lists, with SOURCE taking precedence."
    (let* ((keys
            (cl-loop for (k _) on source by #'cddr collect k))
           (filtered

               for key = (intern (format ":%s" symb))
