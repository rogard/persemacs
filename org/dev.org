  (cl-defun er317/plist-transpose (&key plist)
    "Transpose a PLIST of keys and list-values into a list of row-wise plists.

    Input:
      '(:key1 (v1 v2) :key2 (w1 w2))
    Output:
      ((:key1 v1 :key2 w1)
       (:key1 v2 :key2 w2))"
    (let* ((keys (cl-loop for (k _) on plist by #'cddr collect k))
           (columns (cl-loop for (_ v) on plist by #'cddr collect v))
           (rows (apply #'cl-mapcar #'list columns)))
      (mapcar (lambda (vals)
                (cl-loop for k in keys
                         for v in vals
                         nconc (list k v)))
              rows)))
#+end_src

  (cl-defmacro er317/refactor
      (&key
       source-func
       target-key-list
       result-key-list
       (result-to-plist-p (when target-key-list t))
       target-func
       msg-type)
    "Define TARGET-FUNC as a keyword-based function that wraps around SOURCE-FUNC,
    Zipping to plist its result using TARGET-KEY-LIST if supplied.

         Syntax:
         - SOURCE-FUNC              (quote SYMB) | (lambda (...) ...)
         - <RESULT|TARGET>-KEY-LIST (KEYWORD ... )
         - TARGET-FUNC              (quote SYMB)

         Compile-time:
         - RESULT-FUNC <- er317/zip-to-list or identity

         Runtime:
         TARGET-FUNC (KEYVAL-LIST)
         - RESULT <- (apply SOURCE-FUNC VALUE-LIST)
         - RESULT <- (RESULT-FUNC RESULT)
         - RESULT

         Example:
         ```
         (er317/refactor
         :source-func 'org-babel-get-src-block-info
         :target-func 'er317/org-babel-info-as-plist
         :target-key-list (:no-eval :datum)
         :result-key-list (:lang :body :header-args :switches :name :start :coderef))
         ```"
    (let* (
  	 (result-func
  	  (cond
  	   (result-to-plist-p
  	    `(lambda (result-list)
  	       (er317/zip-to-plist :key-list ',result-key-list :value-list result-list)))
  	   (t #'identity)))
  	 (target-symb
            (cond
             ((symbolp target-func) target-func)
             ((and (consp target-func) (eq (car target-func) 'quote))
              (cadr target-func))
             (t (error "target-func must be a symbol or quoted symbol"))))
     	 (target-symb-list
            (cl-loop for k in target-key-list
                     collect (intern (substring (symbol-name k) 1)))))
      `(cl-defun ,target-symb (&key ,@target-symb-list)
         ,(format "Refactored wrapper around `%s`, accepting keyword args %s."
     		source-func target-key-list)
         (let* ((result-raw (apply ,source-func (list ,@target-symb-list)))
  	      (result (funcall ,result-func result-raw)))
     	 result))))
#+end_src

  (cl-defun er317/filesys-unique-clone
      (&key
       (demo-flag nil)

       (_temp-file
        (when demo-flag
          (let* ((temp-file (make-temp-file nil nil (concat "." "trash")))
                 (text "This file is for demo only; it can be trashed"))
            (with-temp-buffer
              (insert text)
              (write-region (point-min) (point-max) temp-file))
            temp-file)))

       ((source-file)
        (cond
         (demo-flag _temp-file)
         (t (error ":source cannot be nil"))))

       (make-ext (file-name-extension source-file))

       (make-arglist
        (if make-ext
            (list :ext make-ext)
          '(:ignore)))

       (trash-flag t)

       (snapshot-flag nil)

       (_source-content (when snapshot-flag
                          (with-temp-buffer
                            (insert-file-contents source-file)
                            (buffer-string))))
       
       (snapshot-func
        (when snapshot-flag
        	`(lambda (target-file)
             (list
  	    :demo-flag ,demo-flag
  	    :make-arglist ,make-arglist
  	    :source-content ,source-content
  	    :source-exists (file-exists-p ,source-file)
  	    :source-file ,source-file
  	    :target-content (with-temp-buffer (insert-file-contents target-file)(buffer-string))
  	    :target-exists (file-exists-p target-file)
  	    :target-file target-file
  	    :trash-flag ,trash-flag
    	    ))))
       
       (_cleanup-lisp
        `(when (and ,temp-file (file-exists-p ,temp-file))
        	 (delete-file ,temp-file)))
       &allow-other-keys)

    "Clone SOURCE-FILE uniquely.

                  Execution flow:
                  - Generate TARGET-FILE using MAKE-ARGLIST-FULL.
                  - Copy SOURCE-FILE to TARGET-FILE.
                  - If TRASH, trash SOURCE-FILE.
                  - Return TARGET-FILE."
    (unwind-protect
        (let ((target-file
               (apply #'er317/filesys-unique-make make-arglist)))
        	(with-temp-buffer
            (insert-file-contents source-file)
            (write-region (point-min) (point-max) target-file))
        	(message "Cloned source file (%s) to unique file: %s" source-file target-file)
        	(when trash-flag
            (move-file-to-trash source-file)
            (message "Moved source file to trash: %s" source-file))
        	(when snapshot-flag
            (message "Snapshot for %s:\n%s"
                     'er317/filesys-unique-clone
                     (pp-to-string (funcall snapshot-func target-file))))
        	target-file)
      (eval _cleanup-lisp))

    (pcase order

         (append results segment-results)))
